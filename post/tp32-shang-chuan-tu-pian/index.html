￥this -> error = ￥error;

//上传成功，需要拼接文件保存路径，用于添加到数据表
￥data['goods_big_img'] = UPLOAD_PATH . ￥upload_res['savepath'] . ￥upload_res['savename'];

     //商品logo图片上传成功，生成缩略图

  }else{

  }
